**문제2 - 내코드
```c++
//------------------------------------------------------------------------------------------  
// 2023. 2학기 C++ 9월 25일 월요일                         월56목34      (3주 2일)  
//------------------------------------------------------------------------------------------  
// & - reference  
//------------------------------------------------------------------------------------------  
// [과제]  
//------------------------------------------------------------------------------------------  
  
#include <iostream>  
  
using namespace std;  
  
  //내코드
void f(int *a, int *b) {  
    int temp;  
    temp = *a;  
    *a = *b;  
    *b = temp;  
}  
  
int main() {  
  
    int a{1};  
    int b{2};  
    // [문제] 함수 f를 호출하였다. a와 b 값이 교환 되도록 하라.  
    // 함수 호출  
    f(&a, &b);  
  
    cout << a << "," << b << endl; // 화면출력 - 2, 1}
```

**문제2 - 교수님코드

```c++
//------------------------------------------------------------------------------------------  
// 2023. 2학기 C++ 9월 25일 월요일                         월56목34      (3주 2일)  
//------------------------------------------------------------------------------------------  
// & - reference  
//------------------------------------------------------------------------------------------  
// [과제]  
//------------------------------------------------------------------------------------------  
  
#include <iostream>  
  
using namespace std;  
  
int main() {  
  
    int a{1};  
    int b{2};  
    // [문제] 함수 f를 호출하였다. a와 b 값이 교환 되도록 하라. - 선언, 정의 , 호출  
    // 함수 호출  
    void f(int *, int *); //선언  
    f(&a, &b); //호출  
  
    cout << a << "," << b << endl; // 화면출력 - 2, 1}  
  
void f(int *a, int *b) { //정의문  
    int temp;  
    temp = *a;  
    *a = *b;  
    *b = temp;  
}
```

**문제 2 - 개선된 C++ 코드

``` c++
//------------------------------------------------------------------------------------------  
// 2023. 2학기 C++ 9월 25일 월요일                         월56목34      (3주 2일)

void f(int &, int &); //선언  
  
int main() {  
  
    int a{1};  
    int b{2};  
    // [문제] 함수 f를 호출하였다. a와 b 값이 교환 되도록 하라. - 선언, 정의 , 호출  
    // 함수 호출  
    f(a, b); //호출  
  
    cout << a << "," << b << endl; // 화면출력 - 2, 1}  
}  
  
void f(int &a, int &b) { //정의문  
    int temp;  
    temp = a;  
    a = b;  
    b = temp;  
}
```
